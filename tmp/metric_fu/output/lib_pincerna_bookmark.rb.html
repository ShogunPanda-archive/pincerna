53      def filter_and_sort(bookmarks, query)
  
54        # Filtering
  
55        matcher = !query.empty? ? /^#{Regexp.escape(query.downcase)}/i : nil
  
56        bookmarks = bookmarks.select {|bookmark| bookmark[:name] =~ matcher } if matcher
  
57
  
58        # Sorting
  
59        bookmarks.sort {|first, second|
  
60          cmp = first[:name] <=> second[:name]
  
61          cmp = first[:path] <=> second[:path] if cmp == 0
  
62          cmp
  
63        }
  
64      end
