45    # Use data from cache or fetches new data.
  
46    #
  
47    # @param key [String] The key of the data.
  
48    # @param expiration [Fixnum] Expiration of new data, in seconds.
  • Score of 14.35 » flog
  • Complexity 2 » saikuro
 
49    def use(key, expiration)
  
50      value = @data[key]
  
51
  
52      if !value || Time.now.to_f > value[:expiration] then
  
53        data = yield
  
54        @data[key] = {data: data, expiration: Time.now.to_f + expiration}
  
55        data
  
56      else
  
57        value[:data]
  
58      end
  
59    end
  
60  end
