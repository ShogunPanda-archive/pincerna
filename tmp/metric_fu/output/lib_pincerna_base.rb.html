194      def execute_filtering(matches)
  
195        # Get relevant matches and arguments.
  
196        relevant = self.class::RELEVANT_MATCHES
  
197        args = relevant.map {|key, value| value.call(self, matches[key]) }
  
198
  
199        # Now perform the operation
  
200        begin
  
201          perform_filtering(*args)
  
202        rescue => e
  
203          raise e if debug_mode == :error
  
204          []
  
205        end
  
206      end
