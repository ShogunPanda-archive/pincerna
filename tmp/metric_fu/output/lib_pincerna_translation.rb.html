 9  class Translation < Base
  
10    # The expression to match.
  
11    MATCHER = /^
12
  
12        (from\s+)?(?<from>[a-z_-]{2,5})
  
13        \s+
  
14        (to\s+)?((?<to>[a-z_-]{2,5})\s+)?
  
15        (?<text>.+)
  
16      $/mix
  
17
  
18    # Relevant groups in the match.
  
19    RELEVANT_MATCHES = {
  
20      "from" => ->(_, value) { value.downcase },
  
21      "to" => ->(_, value) { value && value.downcase },
  
22      "text" => ->(_, value) { value.strip },
  
23    }
  
24

32    #
  
33    # @param from [String] The code of the source language.
  
34    # @param to [String] The code of the target language.
  
35    # @param value [String] The text to translate.
  
36    # @return [Hash|NilClass] The translation data or `nil` if the translation failed.
  
37    def perform_filtering(from, to, value)
  
38      # By default we translate from English
  
39      if !to then
  
40        to = from
  
41        from = "en"
  
42      end
  
43
  
44      response = Pincerna::Cache.instance.use("translation:#{from}:#{to}:#{value}", Pincerna::Cache::EXPIRATIONS[:short]) {
  
45        fetch_remote_resource(URL, {client: "p", text: value, sl: from, tl: to, multires: 1, ssel: 0, tsel: 0, sc: 1, ie: "UTF-8", oe: "UTF-8"})
  
46      }
  
47
  
48      # Parse results
  
49      if response["dict"] then
  
50        translations = response["dict"][0]["entry"].map {|t| t["word"] }
  
51        {main: translations.shift, alternatives: translations}
  
52      else
  
53        translation = response["sentences"][0]["trans"]
  
54        {main: translation} if translation != value
  
55      end
  
56    end

63      alternatives = results[:alternatives] ? "Alternatives: #{results[:alternatives].join(", ")}" : "Action this item to copy the translation on the clipboard."
  
64      [{title: results[:main], arg: results[:main], subtitle: alternatives, icon: ICON}]
