17    def read_bookmarks
  
18      data = execute_command("/usr/bin/plutil", "-convert", "xml1", "-o", "-", BOOKMARKS_DATA)
  
19
  
20      if data && !data.empty? then
  
21        Plist.parse_xml(data)["Children"].each do |children|

23        end
  
24      end

32      def scan_folder(node, path)
  
33        path += " #{SEPARATOR} #{node["Title"]}"
  
34
  
35        (node["Children"] || []).each do |children|
  
36          children["WebBookmarkType"] == "WebBookmarkTypeLeaf" ? add_bookmark(children["URIDictionary"]["title"], children["URLString"], path) : scan_folder(children, path)
  
37        end
  
38      end
