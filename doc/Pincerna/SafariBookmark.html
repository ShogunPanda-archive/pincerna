def read_bookmarks
  data = execute_command("/usr/bin/plutil", "-convert", "xml1", "-o", "-", BOOKMARKS_DATA)

  if data && !data.empty? then
    Plist.parse_xml(data)["Children"].each do |children|
      scan_folder(children, "")
    end
  end
end
